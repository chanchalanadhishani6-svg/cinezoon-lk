<script>

/* HERO AUTO SLIDER */
const banners=[
  {title:"Sinhala Blockbuster",img:"https://picsum.photos/1600/900?10"},
  {title:"English Action Hit",img:"https://picsum.photos/1600/900?11"},
  {title:"Kids Animation",img:"https://picsum.photos/1600/900?12"}
];
let index=0;
function changeHero(){
  document.getElementById('hero').style.backgroundImage=
  `linear-gradient(rgba(0,0,0,0.4),rgba(0,0,0,0.8)),url(${banners[index].img})`;
  document.getElementById('heroTitle').innerText=banners[index].title;
  index=(index+1)%banners.length;
}
setInterval(changeHero,4000);
changeHero();

function openMovie(title){
  document.getElementById('movieTitle').innerText = title;
  document.getElementById('movieModal').style.display='flex';
}
function closeMovie(){
  document.getElementById('movieModal').style.display='none';
}

function searchMovies(value){
  let cards = document.querySelectorAll('.card');
  cards.forEach(card=>{
    card.style.display = card.innerText.toLowerCase().includes(value.toLowerCase()) ? 'block':'none';
  })
}
