                    if(m.category.includes('Kids')) rows.kids.appendChild(createCard(m));
                });
            }
        }

        function createCard(movie) {
            const card = document.createElement('div');
            card.className = 'movie-card';
            card.onclick = () => openMovieDetail(movie.id);
            
            card.innerHTML = `
                <img src="${movie.img}" alt="${movie.title}" class="card-img" loading="lazy">
                <div class="card-overlay">
                    <div class="card-title">${movie.title}</div>
                    <div class="card-info">
                        <span>${movie.year}</span>
                        <span style="color:#46d369">${movie.match} Match</span>
                    </div>
                </div>
            `;
            return card;
        }

        /* --- INTERACTION HANDLERS --- */
        searchInput.addEventListener('input', (e) => {
            renderMovies(e.target.value);
        });

        function filterByNav(category) {
            // Helper to scroll to sections or filter
            if(category === 'All') {
                window.scrollTo({top: 0, behavior: 'smooth'});
                renderMovies('');
                return;
            }
            
            // Find section ID
            const sectionId = category.toLowerCase();
            const section = document.getElementById(sectionId);
            
            if (section) {
                section.scrollIntoView({ behavior: 'smooth', block: 'start' });
            } else {
                // Fallback search for non-section links like 'Action'
                searchInput.value = category;
                renderMovies(category);
            }
        }

            const movie = movies.find(m => m.id === id) || movies[0]; // Default to first if not found
            
            document.getElementById('mTitle').innerText = movie.title;
            document.getElementById('mMatch').innerText = movie.match + " Match";
            document.getElementById('mYear').innerText = movie.year;
            document.getElementById('mDuration').innerText = movie.duration;
            document.getElementById('mDesc').innerText = movie.desc;
            document.getElementById('modalVideoBg').style.background = `url(${movie.img}) center/cover no-repeat`;

            document.getElementById('movieModal').classList.add('active');
        }

        function toggleAuthMode() {
            const title = document.getElementById('authTitle');
            const submitBtn = document.querySelector('.auth-submit');
            const toggleLink = document.querySelector('.auth-container a:last-of-type');
            
            if (title.innerText === "Sign In") {
                title.innerText = "Register";
                submitBtn.innerText = "Register";
                toggleLink.innerText = "Sign in now";
            } else {
                title.innerText = "Sign In";
                submitBtn.innerText = "Sign In";
                toggleLink.innerText = "Sign up now";
            }
        }
